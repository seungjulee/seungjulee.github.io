import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from IPython.display import Image
df_home = pd.read_csv('./dataset/MLB_STATS_2011_home.csv', error_bad_lines=False)
df_away = pd.read_csv('./dataset/MLB_STATS_2011_away.csv', error_bad_lines=False)
from collections import Counter

increment = 50
width= 25

home_at_bats_bucketted = map(lambda ammt: ammt - ammt%increment, df_home["At-bats"])
away_at_bats_bucketted = map(lambda ammt: ammt - ammt%increment + width, df_away["At-bats"])
home_at_bats_hist = Counter(home_at_bats_bucketted)
away_at_bats_hist = Counter(away_at_bats_bucketted)

minamount = min(min(df_home["At-bats"]), min(df_away["At-bats"]))
maxamount = max(max(df_home["At-bats"]), max(df_away["At-bats"]))
buckets = range(int(minamount), int(maxamount)+1, increment)

fig = plt.figure()
sub = fig.add_subplot(111)
sub.set_title('Histogram (Home At-bats vs Away At-bats)')
sub.bar(home_at_bats_hist.keys(), home_at_bats_hist.values(), color='b', width=width, label="Home At-bats")
sub.bar(away_at_bats_hist.keys(), away_at_bats_hist.values(), color='r', width=width, label="Away At-bats")

sub.legend()
plt.savefig('figures/at_bats_histograms.png', format='png')
Image(filename='figures/at_bats_histograms.png')

from scipy import stats
import pandas as pd
np.random.seed(12345678) # fix random seed to get same numbers

df_away = pd.read_csv('./dataset/MLB_STATS_2011_away.csv', error_bad_lines=False)
# sort by name to compare using paired t-statistics
df_home = df_home.sort(['Team'])
df_away = df_away.sort(['Team'])

print "T-Test p-value (At-bats):", stats.ttest_rel(df_home["At-bats"], df_away["At-bats"]).pvalue 
print "T-Test p-value (Hits):", stats.ttest_rel(df_home["Hits"], df_away["Hits"]).pvalue 
print "T-Test p-value (Runs):", stats.ttest_rel(df_home["Runs"], df_away["Runs"]).pvalue 
print "T-Test p-value (Total Bases):", stats.ttest_rel(df_home["Total Bases"], df_away["Total Bases"]).pvalue
